##Pergunta
12 - Escreve um programa que solicite ao utilizador dois números inteiros e apresente a sua soma.

##Soluções

###Solução 1
using System;
namespace Exercicio12
{	
    class Program
    {		
        //main method
        static void Main(string[] args)
        {
            //declare string variables to be converted
            string a;
            string b;

            //declare the variables to be summed
            int x;
            int y;
            int sum;

            //ask user for a number
            Console.Write("Insert a number: ");

            //converts user input into string
            a = Console.ReadLine();
            x = Convert.ToInt32(a);

            //ask user for a number
            Console.Write("Insert another one: ");

            //converts user input into string
            b = Console.ReadLine();
            y = Convert.ToInt32(b);

            //the sum
            sum = x + y;
            //outputs user's string
            Console.WriteLine(sum);
        }
    }
}
