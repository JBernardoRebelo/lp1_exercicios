##Pergunta
10 - Escreve um programa que solicite ao utilizador uma string e a volte a mostrar no ecrã.

##Soluções

###Solução 1
using System;
namespace Exercicio10
{
    class Program
    {		
        //main method
        static void Main(string[] args)
        {
            //declare variables
            string userInput;

            //ask user for string
            Console.Write("Insert a string: ");

            //converts user input into string
            userInput = Console.ReadLine();

            //outputs user's string
            Console.WriteLine("Your string: " + userInput);
        }
    }
}
