##Pergunta
14 - Escreve um programa que solicite ao utilizador o comprimento, 
altura e largura de um paralelepípedo retângulo, e apresente de seguida o seu volume.

##Soluções

###Solução 1
using System;
namespace Exercicio14
{
    class Program
    {
        static void Main(string[] args)
        {
            //declare variables
            string lenght;
            string height;
            string width;
            //converted variables and result
            double le;
            double he;
            double wi;
            double result;

            //greets user
            Console.WriteLine("This program calculates the");
            Console.WriteLine("volume of a parallelepiped");
            Console.WriteLine("tell me it's Length, Height and Width...");

            //asks user for length
            Console.Write("Insert the length: ");
            //storing and converting to double
            lenght = Console.ReadLine();
            le = Convert.ToDouble(lenght);

            //asks user for height
            Console.Write("Insert the height: ");
            //storing and converting to double
            height = Console.ReadLine();
            he = Convert.ToDouble(height);

            //asks user for width
            Console.Write("Insert the width: ");
            //storing and converting to double
            width = Console.ReadLine();
            wi = Convert.ToDouble(width);

            //calculates result using the formula
            result = le * he * wi;

            //outputs result to user
            Console.WriteLine("The volume is " + result);
        }
    }
}
