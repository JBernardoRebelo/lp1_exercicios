##Pergunta
11 - Escreve um programa que solicite ao utilizador uma string e a volte a mostrar no ecrã de forma invertida (isto é, de trás para a frente).

##Soluções

###Solução 1
using System;
namespace Exercicio10
{	
    class Program
    {		
        //main method
        static void Main(string[] args)
        {
            //declare variables
            string userInput;
            char [] usersArray;

            //ask user for string
            Console.Write("Insert a string to be reversed: ");

            //stores user input as string
            userInput = Console.ReadLine();

            //converts user's string into a char array
            usersArray = userInput.ToCharArray();

            //reverses the order of said array
            Array.Reverse(usersArray);

            //outputs chars in the array
            Console.WriteLine(usersArray);
        }
    }
}
